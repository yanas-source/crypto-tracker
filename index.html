        async function loadTransactions() {

            const network = document.getElementById('network').value;

            const address = WALLETS[network];

            const txList = document.getElementById('tx-list');

            const loading = document.getElementById('loading');

            

            document.getElementById('wallet-address').innerText = address;

            txList.innerHTML = '';

            loading.style.display = 'block';


            try {

                if (network === 'tron') {

                    await fetchTronTx(address);

                } else if (network === 'ethereum') {

                    await fetchEthTx(address);

                }

            } catch (error) {

                loading.innerText = 'Помилка при завантаженні транзакцій.';

                console.error(error);

            } finally {

                if (txList.children.length > 0) {

                    loading.style.display = 'none';

                }

            }

        }


        // Отримання транзакцій TRON через TronScan API

        async function fetchTronTx(address) {

            const response = await fetch(`https://apilist.tronscan.org/api/transaction?sort=-timestamp&limit=10&start=0&address=${address}`);

            const data = await response.json();

            const txs = data.data;


            const txList = document.getElementById('tx-list');

            

            txs.forEach(tx => {

                const li = document.createElement('li');

                const isIncoming = tx.toAddress.toLowerCase() === address.toLowerCase();

                const tag = isIncoming 

                    ? '<span class="tag tag-in">ВХІДНА</span>' 

                    : '<span class="tag tag-out">ВИХІДНА</span>';

                

                // Формування повного посилання на TronScan

                const fullUrl = `https://tronscan.org/#/transaction/${tx.hash}`;


                li.innerHTML = `

                    <div>

                        ${tag}

                        <a href="${fullUrl}" target="_blank" rel="noopener noreferrer">${fullUrl}</a>

                    </div>

                `;

                txList.appendChild(li);

            });

        }


        // Отримання транзакцій Ethereum через публічний API Blockchair (не вимагає ключів)

        async function fetchEthTx(address) {

            const response = await fetch(`https://api.blockchair.com/ethereum/dashboards/address/${address}?limit=10`);

            const data = await response.json();

            const txHashes = data.data[address.toLowerCase()].calls;


            const txList = document.getElementById('tx-list');


            txHashes.forEach(call => {

                const li = document.createElement('li');

                

                // Формування повного посилання на Etherscan

                const fullUrl = `https://etherscan.io/tx/${call.transaction_hash}`;


                li.innerHTML = `

                    <div>

                        <a href="${fullUrl}" target="_blank" rel="noopener noreferrer">${fullUrl}</a>

                    </div>

                `;

                txList.appendChild(li);

            });

        }
